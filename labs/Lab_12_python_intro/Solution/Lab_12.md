```python
with open('input.txt', 'r') as file:
    N, M = map(int, file.readline().split())
    if not (1 <= N <= 50 and 1 <= M <= 50):
        print("–ù—É–∂–Ω—ã —á–∏—Å–ª–∞ –æ—Ç 1 –¥–æ 50!! –∏–∑–º–µ–Ω–∏—Ç–µ —á–∏—Å–ª–∞ –≤ –≤—Ö–æ–¥–Ω–æ–º —Ñ–∞–π–ª–µ –∏ –ø–µ—Ä–µ–∑–∞–ø—É—Å—Ç–∏—Ç–µ –∫–æ–¥")

    dp = [[0] * M for _ in range(N)]
    dp[0][0] = 1

    for i in range(N):
        for j in range(M):
            if i + 1 < N and j + 2 < M:
                dp[i+1][j+2] += dp[i][j]
            if i + 2 < N and j + 1 < M:
                dp[i+2][j+1] += dp[i][j]

    result = dp[N-1][M-1]
    print(result)
```

```python
with open('input.txt', 'r') as file:
```

```python
with open('input.txt', 'r') as file:
    text = file.read()
    char_count = {}
    for char in text:
        if char not in char_count and char != ' ' and char != '\n':
            char_count[char] = 1
        elif char != ' ' and char != '\n':
            char_count[char] += 1

    sorted_chars = sorted(char_count.keys())
    max_count = max(char_count.values())

    for i in range(max_count, 0, -1):
        line = ''
        for char in sorted_chars:

            else:
                line += ' '
        print(line)
```
